\begin{frame}{Exemple : $(a^+ | b^+) c^\star$}

  \tfBlock[top=-3mm]{Représentation mathématique}{
    $$\begin{array}{lccl}
      \text{\structure{alphabet}} & \Sigma & = & \{a, b, c\}\\
      \text{\structure{états}} & Q & = & \{s_1, s_2, s_3, s_4, s_5\}\\
      \text{\structure{états initiaux}} & I & = & \{s_1, s_4\}\\
      \text{\structure{états finaux}} & F & = & \{s_3\}\\
      \text{\structure{transitions}} & \mu & = & \left\{ \begin{array}{lll}
        
        \langle s_1, a, s_1 \rangle,&
        \langle s_1, a, s_2 \rangle,&
        \langle s_2, a, s_2 \rangle,\\

        \langle s_4, b, s_5 \rangle,&
        \langle s_4, b, s_4 \rangle,&
        \langle s_5, b, s_5 \rangle,\\

        \langle s_2, \varepsilon, s_3 \rangle,&
        \langle s_5, \varepsilon, s_3 \rangle,&
        \langle s_3, c, s_3 \rangle
      \end{array}\right\}\\
    \end{array}$$
  }

  \tfBlock[left=.5\textwidth, bottom]{Représentation graphique}{\footnotesize
      \begin{tikzpicture}[smAutomaton,node distance=1.5cm]
        \smState[\smInitial]   (s1) at (0.0,1)  {$s_1$}; 
        \smState               (s2) at (1.5,1)  {$s_2$}; 
        \smState[\smAccepting] (s3) at (3.0,.5) {$s_3$}; 
        \smState[\smInitial]   (s4) at (0.0,0)  {$s_4$}; 
        \smState               (s5) at (1.5,0)  {$s_5$}; 

        \smPath (s1) edge             node[swap]  {$a$}           (s2);
        \smPath (s1) edge[loop above] node        {$a$}           (s1);
        \smPath (s2) edge[loop above] node        {$a$}           (s2);
        \smPath (s4) edge             node[swap]  {$b$}           (s5);
        \smPath (s4) edge[loop below] node        {$b$}           (s4);
        \smPath (s5) edge[loop below] node        {$b$}           (s5);
        \smPath (s3) edge[loop right] node        {$c$}           (s3);
        \smPath (s2) edge             node        {$\varepsilon$} (s3);
        \smPath (s5) edge             node[swap]  {$\varepsilon$} (s3);
    \end{tikzpicture}
  }


  \tfBlock[right=.5\textwidth, bottom]{Représentation matricielle}{
      \begin{tabular}{|l|c|c|c|c|}

        $s_4$ i& &$s_4$, $s_5$  & & \\
        $s_5$ &  & $s_5$ & &$s_3$ \\
        \hline
      \end{tabular}
  }
