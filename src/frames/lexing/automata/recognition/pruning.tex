\SetKwFunction{Emondage}{emondage}
\SetKwFunction{Accessible}{accessible}
\SetKwFunction{Coaccessible}{coaccessible}

\begin{frame}[fragile]{Algorithme de l'Ã©mondage}

  \tf[text, top]{
    \begin{algorithm}[H]
      \Fn{\Emondage($A = \langle \Sigma, Q, I, F, \rightarrow \rangle$ : automate) : automate}{
        \tfStructure<2|handout>{$\Accessible \leftarrow I$}\;
        \Tantque{\tfStructure<3-5|handout>{$\exists a \in \Sigma, \exists q\in \Accessible, \exists q'\notin\Accessible, q \xrightarrow{a} q'$}}{
          \tfStructure<3-5|handout>{$\Accessible \leftarrow \Accessible \cup \{q'\}$}\;
        }
        \tfExample<6|handout>{$\Coaccessible \leftarrow F$}\;
        \Tantque{\tfExample<7-9|handout>{$\exists a \in \Sigma, \exists q\notin \Coaccessible, \exists q'\in\Coaccessible, q \xrightarrow{a} q'$}}{
          \tfExample<7-9|handout>{$\Coaccessible \leftarrow \Coaccessible \cup \{q\}$}\;
        }
        \tfAlert<10|handout>{$Q' \leftarrow \Accessible \cap \Coaccessible$}\;
        \Retourner $\displaystyle\left\langle \Sigma, Q', I \cap Q', F \cap Q', \rightarrow \cap (Q' \times \Sigma \times Q')\right\rangle$\;
      }
    \end{algorithm}
  }

  \tfExampleBlock[bottom]{Exemple}{

    \item<10-|handout> \alert{$Q' = \{2, 3, 6\}$}
