\begin{frame}{Stabilité par complémentaire}
  
  \tfBlock[top]{Théorème -- Stabilité par complémentaire}{
    Soient $\Sigma$ un alphabet et $L \in \textsc{rat}_\Sigma$. Alors $\overline{L} = \Sigma^\star \setminus L \in \textsc{rat}_\Sigma$. 
  }

  \tfBlock<2-|handout>[y=8mm]{Preuve}{
    \begin{itemize}
    \item Il existe $\example{A = \langle \Sigma, Q, I, F, \rightarrow \rangle}$ \alert{déterministe complet} qui reconnaît $L$.
    \item $\overline{L}$ est reconnu par $\structure{\overline{A} = \langle \Sigma, Q, I, Q \setminus F, \rightarrow \rangle}$.
    \end{itemize}
  }

  \tfExampleBlock[y=-10mm]{Exemple}{}

  \tf[bottom,x=-.25\textwidth]{
    \begin{tikzpicture}[smAutomaton, example]
      \node at (1.0,1.3) {$\{ u\in \{a, b\}^\star \;|\; |u|_a \equiv 1 \pmod{2} \}$};
      \smState[\smInitial]   (a) at (0.0,0.0) {$0$};
      \smState[\smAccepting] (b) at (2.0,0.0) {$1$};
      
      \smPath (a) edge[bend left ] node {$a$} (b);
      \smPath (b) edge[bend left ] node {$a$} (a);
      \smPath (a) edge[loop below] node {$b$} (a);
      \smPath (b) edge[loop below] node {$b$} (b);
    \end{tikzpicture}
  }
  
  \tf<2-|handout>[bottom,x=.25\textwidth]{
    \begin{tikzpicture}[smAutomaton, structure]
      \node at (1.0,1.3) {$\{ u\in \{a, b\}^\star \;|\; |u|_a \equiv 0 \pmod{2} \}$};
      \smState[\smInitial\smAccepting]   (a) at (0.0,0.0) {$0$};
      \smState                           (b) at (2.0,0.0) {$1$};
      
      \smPath (a) edge[bend left ] node {$a$} (b);
      \smPath (b) edge[bend left ] node {$a$} (a);
      \smPath (a) edge[loop below] node {$b$} (a);
      \smPath (b) edge[loop below] node {$b$} (b);
    \end{tikzpicture}
  }
  
\end{frame}
