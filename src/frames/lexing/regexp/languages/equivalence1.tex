\begin{frame}{Équivalence entre expressions rationnelles}

  \begin{block}{Définition -- Équivalence entre expressions rationnelles}
    Deux expressions rationnelles \alert{$u$} et \alert{$v$} sont dites \structure{équivalentes}
    (ou par abus de langage ``égales''), noté \alert{$u \equiv v$},
    si elles décrivent le même langage.
    \alert{$$u \equiv v ~\eqdef~ \mathcal{S}(u) = \mathcal{S}(v)$$}
  \end{block}

  \begin{exampleblock}{Exemples}
    \begin{itemize}
    \item \example{$a^\star | b^\star \not\equiv (a|b)^\star$} : $ab \in \mathcal{S}((a|b)^\star)$ mais $ab \not\in \mathcal{S}(a^\star | b^\star)$
    \item \example{$(a^\star | b^\star)^\star \equiv (a|b)^\star$} : $\mathcal{S}((a^\star | b^\star)^\star) = \mathcal{S}((a|b)^\star) = \{a, b\}^\star$
    \end{itemize}
  \end{exampleblock}
  
  
\end{frame}
