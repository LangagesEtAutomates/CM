\begin{frame}{Équivalence grammaire rationnelle / automate}

  \tfBlock[top]{Théorème}{
    Soit $\Sigma$ un alphabet et $L \in \mathscr{P}(\Sigma^\star)$ un langage. \\
    $L$ est généré par une grammaire rationnelle à droite ssi L est rationnel. 
  }

  \tfBlock[y=0mm]{Démonstration}{
    \begin{itemize}
    \item $L$ est engendré par la grammaire rationnelle à droite \structure{$G = \langle \Sigma, \Gamma, S, \rightarrow_G \rangle$}
      \footnote[frame,1]{on suppose $G$ sans règle $\structure{A \rightarrow a}$, qui peut toujours être remplacé par \structure{$A \rightarrow a \cdot F$}, avec \structure{$F \rightarrow \varepsilon$}}
    \item ssi, $L$ est reconnu par l'AFN unitaire $\varepsilon$-libre \structure{$A = \langle \Sigma, \Gamma, \{S\}, F, \rightarrow_A \rangle$}, avec
    \end{itemize}
    $$\structure{\begin{array}{r@{~}c@{~}l@{\quad\Leftrightarrow\quad}r@{~}c@{~}l}
        X & \xrightarrow{a}_A & Y & X & \rightarrow_G & a Y \\
        X & \in               & F & X & \rightarrow_G & \varepsilon 
    \end{array}}$$
  }

  \tfExampleBlock[y=-25mm]{Exemple}{\vspace{-6mm}
    $$\left\{\begin{array}{@{\,}r@{~\rightarrow~}l@{\,\mid\,}l@{\,}}
        S & aA\\
        A & aA & bB\\
        B & bB & \varepsilon
      \end{array}\right.
    \quad\quad\Leftrightarrow\quad\quad
    \begin{tikzpicture}[smAutomaton, baseline=(S)]
      \smState[\smInitial]   (S) at (0.0, 0) {$S$}; 
      \smState               (A) at (1.5, 0) {$A$}; 
      \smState[\smAccepting] (B) at (3.0, 0) {$B$}; 
      \smPath (S) edge             node {$a$} (A);
      \smPath (A) edge[loop above] node {$a$} (A);
      \smPath (A) edge             node {$b$} (B);
      \smPath (B) edge[loop above] node {$b$} (B);
    \end{tikzpicture}
    $$
  }

\end{frame}
