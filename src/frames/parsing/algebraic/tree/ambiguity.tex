\begin{frame}{Ambiguité d'une grammaire algébrique}

  \tfBlock[top]{Définition -- Ambiguïté}{
    Une grammaire est \structure{ambiguë} si \alert{un mot a plusieurs arbres de dérivation}.
  }

  \tfExampleBlock[y=10mm]{Exemple}{
    \begin{itemize}
    \item $u = a \oplus a \otimes a$
    \item $G = \{A \rightarrow A \oplus A \mid A \otimes A \mid a\}$
    \end{itemize}
  }

  \tf[bottom, x=-.25\textwidth]{
    \begin{tikzpicture}
      \node[structure] (a) at (3,3) {$A$};        
      \node[structure] (b) at (1,2) {$A$};        \smPath (a) edge (b);
      \node[structure] (c) at (0,1) {$A$};        \smPath (b) edge (c);
      \node[structure] (d) at (2,1) {$A$};        \smPath (b) edge (d);
      \node[structure] (e) at (4,2) {$A$};        \smPath (a) edge (e);
      \node[alert]     (f) at (0,0) {$a$};        \smPath (c) edge (f);
      \node[alert]     (g) at (1,1) {$\oplus$};   \smPath (b) edge (g);
      \node[alert]     (h) at (2,0) {$a$};        \smPath (d) edge (h);
      \node[alert]     (i) at (3,2) {$\otimes$};  \smPath (a) edge (i);
      \node[alert]     (j) at (4,1) {$a$};        \smPath (e) edge (j);
    \end{tikzpicture}
  }
