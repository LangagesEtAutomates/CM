\begin{frame}{Ambiguïté d'une grammaire algébrique}

  \tfBlock[top=-3mm]{Définition -- Ambiguïté}{
    Une grammaire est \structure{ambiguë} si \alert{un mot a plusieurs arbres de dérivation}.
  }

  \tfExampleBlock[y=16mm]{Exemple : $u = a + a \times a$ et $G = \{A \rightarrow A + A \mid A \times A \mid a\}$}{}

  \tf[y=-2mm,x=-.25\textwidth]{
    \begin{tikzpicture}
      \node[example]       at (2.4,2.5) {$(a + a) \times a$};

      \node[structure] (a) at (2.4,2.1) {$A$};        
      \node[structure] (b) at (0.8,1.4) {$A$};      \smPath (a) edge (b);
      \node[structure] (c) at (0.0,0.7) {$A$};      \smPath (b) edge (c);
      \node[structure] (d) at (1.6,0.7) {$A$};      \smPath (b) edge (d);
      \node[structure] (e) at (3.2,1.4) {$A$};      \smPath (a) edge (e);
      \node[alert]     (f) at (0.0,0.0) {$a$};      \smPath (c) edge (f);
      \node[alert]     (g) at (0.8,0.7) {$+$};      \smPath (b) edge (g);
      \node[alert]     (h) at (1.6,0.0) {$a$};      \smPath (d) edge (h);
      \node[alert]     (i) at (2.4,1.4) {$\times$}; \smPath (a) edge (i);

  \tfBlock[bottom=-1mm]{Remarques}{
    \vspace{-2mm}
    \begin{itemize}
    \item L'ambiguïté est une \structure{propriété de la grammaire}, pas du langage
    \item Il existe des \structure{langages intrinsèquement ambigus}
      \begin{itemize}
      \item Toutes les grammaires qui l'engendrent sont ambiguës
      \item Par exemple \example{$\left\{ a^i b^j c^k \,\middle\mid\, i=j \lor j=k \right\}$}
      \end{itemize}
    \end{itemize}
  }
